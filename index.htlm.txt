        function updateDashboard() {
            const currentData = data[currentCategory];
            const tableBody = document.getElementById('simulacros-body');
            tableBody.innerHTML = '';

            if(currentData.length === 0) {
                tableBody.innerHTML = '<tr><td colspan="5" class="text-center py-12 text-slate-400 text-sm font-medium">No hay simulacros registrados.</td></tr>';
                document.getElementById('stat-avg').innerText = '0.0%';
                document.getElementById('stat-last').innerText = '0';
                document.getElementById('stat-prob').innerText = '0%';
                return;
            }

            let totalPct = 0;
            currentData.forEach(s => {
                const pct = ((s.correct / s.total) * 100).toFixed(1);
                totalPct += parseFloat(pct);
                
                tableBody.innerHTML += `
                    <tr class="hover:bg-slate-50 transition-all">
                        <td class="px-4 py-4">
                            <p class="font-bold text-slate-700">${s.name}</p>
                            <p class="text-[10px] text-slate-400 font-bold uppercase">${s.date}</p>
                        </td>
                        <td class="px-4 py-4 text-sm font-black">${s.correct}/${s.total}</td>
                        <td class="px-4 py-4">
                            <span class="text-xs font-black ${pct >= 75 ? 'text-blue-600' : 'text-orange-600'}">
                                ${pct}%
                            </span>
                        </td>
                        <td class="px-4 py-4">
                            <div class="flex items-center gap-2">
                                <div class="w-16 bg-slate-100 h-1.5 rounded-full overflow-hidden">
                                    <div class="h-full ${pct >= 75 ? 'bg-blue-500' : 'bg-orange-400'}" style="width: ${pct}%"></div>
                                </div>
                            </div>
                        </td>
                        <td class="px-4 py-4 text-right">
                            <button onclick="deleteEntry(${s.id})" class="text-slate-300 hover:text-red-500 transition-all"><i class="fas fa-trash"></i></button>
                        </td>
                    </tr>
                `;
            });

                            total: parseInt(row.Total || row['N° Preg'] || 100),
                            date: new Date().toLocaleDateString('es-PE', { day:'2-digit', month:'short' }),
                            breakdown: { Ciru: 50, Pedia: 50, Gineco: 50, Med: 50 }
                        });
                    });
                    saveAndRefresh();
                    toggleUploadModal();
                };
                reader.readAsArrayBuffer(file);
            });
        }

        // --- Charts Section ---
        function initCharts() {
            const mCtx = document.getElementById('mainChart').getContext('2d');
            mainChart = new Chart(mCtx, {
                type: 'line',
                data: { labels: [], datasets: [{ data: [], borderColor: '#3b82f6', tension: 0.4, fill: true, backgroundColor: 'rgba(59, 130, 246, 0.05)', borderWidth: 3 }] },
                options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { y: { beginAtZero: true, max: 100 } } }
            });

            const rCtx = document.getElementById('radarChart').getContext('2d');
            radarChart = new Chart(rCtx, {
                type: 'radar',
                data: { labels: ['Ciru', 'Pedia', 'Gineco', 'Med', 'Infec'], datasets: [{ data: [0,0,0,0,0], borderColor: '#3b82f6', backgroundColor: 'rgba(59, 130, 246, 0.2)' }] },
                options: { scales: { r: { suggestedMin: 0, suggestedMax: 100, ticks: { display: false } } }, plugins: { legend: { display: false } } }
            });
        }

        function updateCharts(currentData) {
            mainChart.data.labels = currentData.map(s => s.name);
            mainChart.data.datasets[0].data = currentData.map(s => (s.correct/s.total)*100);
            mainChart.update();

            const last = currentData[currentData.length-1];
            if(last.breakdown) {
                radarChart.data.datasets[0].data = [last.breakdown.Ciru, last.breakdown.Pedia, last.breakdown.Gineco, last.breakdown.Med, 45];
                radarChart.update();
            }
        }

        // --- AI Strategic Analysis ---
        async function runAIAnalysis() {
            const stats = data[currentCategory];
            if(stats.length === 0) return alert("Sube simulacros primero.");

            const btn = document.getElementById('ai-btn');
            const loader = document.getElementById('ai-loader');
            const btnText = document.getElementById('ai-btn-text');
            const insightBox = document.getElementById('ai-insight-text');

            btn.disabled = true;
            loader.classList.remove('hidden');
            btnText.innerText = "Consultando Estrategia...";

            const summary = stats.map(s => `${s.name}: ${s.correct}/${s.total}`).join(", ");
            const masteredCount = Object.values(masteredTopics).filter(v => v).length;
            
            const prompt = `Analiza mi rendimiento: ${summary}. He dominado ${masteredCount} temas clave. Dame un consejo de una frase y 3 puntos tácticos para subir mi puntaje esta semana. Usa un tono de mentor médico.`;

            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
                });

                const result = await response.json();
                insightBox.innerHTML = result.candidates?.[0]?.content?.parts?.[0]?.text.replace(/\n/g, '<br>') || "Error al procesar.";
            } catch (e) {
                insightBox.innerText = "Error de conexión con la IA.";
            } finally {
                btn.disabled = false;
                loader.classList.add('hidden');
                btnText.innerText = "Actualizar Análisis";
            }
        }
